#include <stdio.h>
#include <ctype.h>

int main() {
    char c;
    int characters = 0, words = 0, lines = 0;
    int inWord = 0;

    printf("Enter text (Ctrl+D to end):\n");

    while ((c = getchar()) != EOF) {
        characters++;

        if (c == '\n')
            lines++;

        if (isspace(c))
            inWord = 0;
        else if (!inWord) {
            inWord = 1;
            words++;
        }
    }

    printf("Characters: %d\n", characters);
    printf("Words: %d\n", words);
    printf("Lines: %d\n", lines);

    return 0;
}
