%{
#include <stdio.h>
#include <stdlib.h>

int num1, num2;
%}

%%
[0-9]+[\+][0-9]+   {
                        sscanf(yytext, "%d+%d", &num1, &num2);
                        printf("%d + %d = %d\n", num1, num2, num1+num2);
                    }

[0-9]+[-][0-9]+    {
                        sscanf(yytext, "%d-%d", &num1, &num2);
                        printf("%d - %d = %d\n", num1, num2, num1-num2);
                    }

[0-9]+[\*][0-9]+   {
                        sscanf(yytext, "%d*%d", &num1, &num2);
                        printf("%d * %d = %d\n", num1, num2, num1*num2);
                    }

[0-9]+[/][0-9]+    {
                        sscanf(yytext, "%d/%d", &num1, &num2);
                        if(num2 == 0)
                            printf("%d / %d = Error (Division by zero)\n", num1, num2);
                        else
                            printf("%d / %d = %d\n", num1, num2, num1/num2);
                    }

[ \t\n]+   ;   // ignore whitespace
.          ;   // ignore other chars
%%

int yywrap() { return 1; }

int main() {
    printf("Enter arithmetic expressions (like 12+5, 8*3, 9-2, 10/2):\n");
    yylex();
    return 0;
}
